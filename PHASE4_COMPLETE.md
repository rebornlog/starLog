**板块识别算法：**
```typescript
// 金融：银行、保险、证券
if (code.startsWith('601') || name.includes('银行')) return 'finance'

// 科技：电子、信息、软件
if (code.startsWith('002') || name.includes('科技')) return 'tech'

// 消费：食品、饮料、酒
if (name.includes('酒') || name.includes('食品')) return 'consumer'

// 医药：药、医疗、生物
if (name.includes('药')) return 'medicine'

// 能源：石油、石化、电力
if (name.includes('石油') || name.includes('电力')) return 'energy'

// 制造：制造、机械、汽车
if (name.includes('制造') || name.includes('汽车')) return 'manufacturing'
```
